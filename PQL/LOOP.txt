DECLARE
	I NUMBER(3);
BEGIN
	I:=1;
LOOP
	DBMS_OUTPUT.PUT_LINE(I);
	EXIT WHEN I>=100;
	I:=I+1;
END LOOP;
END;

/DECLARE 
	CURSOR TIPOS_AA IS 

	SELECT *  
	FROM TCLIENTE
	WHERE TIPO='AA';

TIPAA TIPOS_AA%ROWTYPE;

BEGIN

	OPEN TIPOS_AA;
LOOP 
	FETCH TIPOS_AA INTO TIPAA;
	DBMS_OUTPUT.PUT_LINE(TIPAA.NOMBRE);
	EXIT WHEN  TIPOS_AA% NOTFOUND;
END LOOP;
CLOSE TIPOS_AA;
	END;
/
