CREATE OR REPLACE PROCEDURE NOMfECHAEMPLE  IS


CURSOR EMPLE IS
	SELECT NOMBRE,FECHA_ALTA
	FROM EMPLEADO 
	ORDER BY NOMBRE;

FILAEMP EMPLE%ROWTYPE;

BEGIN 

	OPEN EMPLE;
		
		LOOP
			FETCH EMPLE INTO FILAEMP;
			EXIT WHEN EMPLE%NOTFOUND;
			DBMS_OUTPUT.PUT_LINE('NOMBRE: ' || FILAEMP.NOMBRE || 'FECHA DE ALTA: ' ||  FILAEMP.FECHA_ALTA);
	
		END LOOP;

	CLOSE EMPLE;

END;
