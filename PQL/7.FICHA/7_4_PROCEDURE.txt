CREATE OR REPLACE PROCEDURE EMPLEADO_MAX_SALARIO IS

CURSOR  CEMPLEADO IS
    SELECT *
    FROM EMPLEADO
    ORDER BY SALARIO ASC;

FILAEMPLEADO CEMPLEADO%Rowtype;

BEGIN
OPEN CEMPLEADO;
    LOOP

        FETCH CEMPLEADO INTO FILAEMPLEADO;
        EXIT WHEN CEMPLEADO%NOTFOUND OR CEMPLEADO%ROWCOUNT >5;
          DBMS_OUTPUT.PUT_LINE(FILAEMPLEADO.NOMBRE);
        END LOOP;

CLOSE CEMPLEADO;







END;
