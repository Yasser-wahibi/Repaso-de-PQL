CREATE OR REPLACE FUNCTION SOLOLETRAS (CADENA VARCHAR2) 
RETURN VARCHAR2 IS
    RESUL VARCHAR2(100) := '';
BEGIN
    FOR I IN 1..LENGTH(CADENA) LOOP
        IF (SUBSTR(CADENA, I, 1) BETWEEN 'A' AND 'Z' OR SUBSTR(CADENA, I, 1) BETWEEN 'a' AND 'z') THEN
            RESUL := RESUL || SUBSTR(CADENA, I, 1);
        ELSE 
            RESUL := RESUL || ' ';
        END IF;
    END LOOP;
    RETURN RESUL;
END;
/
