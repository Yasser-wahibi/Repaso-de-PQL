CREATE OR REPLACE PROCEDURE CAMBIOLOC(NUM DEPARTAMENTO.NUM_DEPTO%TYPE, CADENA DEPARTAMENTO.LOCALIDAD%TYPE) is

    filadep DEPARTAMENTO%ROWTYPE;
BEGIN

    SELECT * INTO filadep 
    FROM DEPARTAMENTO 
    WHERE NUM_DEPTO = NUM;


    UPDATE DEPARTAMENTO 
    SET LOCALIDAD = CADENA 
    WHERE NUM_DEPTO = NUM;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('El codigo de num_depto no existe');
END;
/
