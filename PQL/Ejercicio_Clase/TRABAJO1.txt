CREATE or replace FUNCTION fun1 (NOM MAT_ART.NOM_MAT%TYPE)
return number is 

     CONT NUMBER(5);
     FILMAT MAT_ART%ROWTYPE;
BEGIN
   SELECT * INTO FILMAT FROM MATERIA WHERE NOM= NOMBRE;
    SELECT COUNT(*) INTO CONT FROM MAT_ART WHERE NOM= NOM_MAT;
RETURN CONT; 

EXCEPTION
    WHEN NO_DATA_FOUND THEN 
    DBMS_OUTPUT.PUT_LINE('L AMETRIA NO EXISTE');
    return null;
        

END;
/
