DECLARE
    CURSOR EMPLEDEP IS
        SELECT NOMBRE
        FROM EMPLEADO
        WHERE CONTARBLANCOS(NOMBRE) > 2;
    
    EMPLE EMPLEDEP%ROWTYPE;

BEGIN
    OPEN EMPLEDEP;

    LOOP
        FETCH EMPLEDEP INTO EMPLE;
        EXIT WHEN EMPLEDEP%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(EMPLE.NOMBRE);
    END LOOP;

    CLOSE EMPLEDEP;
END;
/
