CREATE OR REPLACE TRIGGER MODIF_PRO  AFTER  UPDATE OF NOMBRE,TOTAL_COMPRA  ON  TPROVINCIA FOR EACH ROW


BEGIN 

    IF UPDATING('NOMBRE') THEN
        DBMS_OUTPUT.PUT_LINE('SE HA MODIFICADO EN LA TABLA TPOVINCIA: ' || :OLD.NOMBRE || '  QUE PASA AS LLAMARSE  '|| :NEW.NOMBRE);
    END IF;

    IF UPDATING ('TOTAL_COMPRA') THEN
            DBMS_OUTPUT.PUT_LINE('SE HA MODIFICADO EN LA TABLA TPOVINCIA: ' || :OLD.TOTAL_COMPRA || '    Y SU VALOR NUEVO ES    '|| :NEW.TOTAL_COMPRA);
    END IF;

END;
/
