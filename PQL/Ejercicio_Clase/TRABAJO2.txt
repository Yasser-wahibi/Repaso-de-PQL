CREATE OR REPLACE FUnCTION FUN2 (NOM PROYECTO.NOMBRE%TYPE )
return boolean is
    CON NUMBER(4);
    filapro proyecto%ROWTYPE;
BEGIN

SELECT * into filapro FROM PROYECTO WHERE NOM=NOMBRE; 

SELECT COUNT(*) INTO CON FROM ARTICULO WHERE NOM=NOM_PROY;

IF CON>0 THEN 
    RETURN TRUE;

ELSE
    RETURN FALSE;

END IF;


EXCEPTION 

    WHEN  NO_DATA_FOUND THEN 
    DBMS_OUTPUT.pUT_LINE('NO SE HA ENCONOTRADO EL PROYECTO');
        RETURN NULL;
END;
