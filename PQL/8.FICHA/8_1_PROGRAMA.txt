ACCEPT DNI PROMPT 'ESCRIBA SU DNI'; 
DECLARE
CARNET TCLIENTE.DNI%TYPE:='&DNI';
FILACLI TCLIENTE%Rowtype;

BEGIN
    SELECT * INTO FILACLI FROM TCLIENTE
    
    WHERE DNI=CARNET;
    UPDATE TCLIENTE SET TIPO='EE' WHERE DNI =CARNET;
EXCEPTION 
    WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE("ESE CLIENTE NO EXISTE");


END;
/
