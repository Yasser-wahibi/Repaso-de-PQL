DECLARE CURSOR TVEN IS
	SELECT SUPERIOR, COUNT(*) CONT
	FROM TVENDEDOR
	GROUP BY SUPERIOR;
     
    DATOS TVEN%ROWTYPE;


    CONT_001 NUMBER := 0;
    CONT_002 NUMBER := 0;
    CONT_003 NUMBER := 0;
    CONT_NULL NUMBER := 0;

BEGIN
    OPEN TVEN;

    LOOP
        FETCH TVEN INTO DATOS;
        EXIT WHEN TVEN%NOTFOUND;
	DBMS_OUTPUT.PUT_LINE(DATOS.CONT || 'VENDORE TIENE COMO SUPERIOR A ' ||DATOS.SUPERIOR);
    END LOOP;


   
    CLOSE TVEN;
END;
/
