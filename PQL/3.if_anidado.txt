ACCEPT NUM1 PROMPT 'ESCRIBE UN NUMERO: '
ACCEPT NUM2 PROMPT 'ESCRIBE UN NUMERO: '

DECLARE
  NUM1 NUMBER(5) := &NUM1;
  NUM2 NUM1%TYPE := &NUM2;
BEGIN
  IF NUM1 > NUM2 THEN 
    DBMS_OUTPUT.PUT_LINE(NUM1 || ' ES MAYOR QUE ' || NUM2);
  ELSIF NUM1 < NUM2 THEN
    DBMS_OUTPUT.PUT_LINE(NUM1 || ' ES MENOR QUE ' || NUM2);
  ELSE
    DBMS_OUTPUT.PUT_LINE(NUM1 || ' ES IGUAL A ' || NUM2);
  END IF;
END;
/
