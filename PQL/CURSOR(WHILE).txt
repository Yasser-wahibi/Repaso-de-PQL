DECLARE
	CURSOR EMP_ANALISTA IS
	SELECT NUM_EMP,NOMBRE,NUM_DEP
	FROM EMPLEADO
	WHERE PUESTO='ANALISTA'
	ORDER BY NOMBRE ASC;

EMPLEADO_AN EMP_ANALISTA%ROWTYPE;

BEGIN
	OPEN EMP_ANALISTA;
	FETCH EMP_ANALISTA INTO EMPLEADO_AN;
	WHILE EMP_ANALISTA%FOUND LOOP
	
	DBMS_OUTPUT.PUT_LINE('NOMBRE:' ||EMPLEADO_AN.NOMBRE || ' NUM_EMP: ' || EMPLEADO_AN.NUM_EMP || 'DEPATAMENTO: ' || EMPLEADO_AN.NUM_DEP );
	FETCH EMP_ANALISTA INTO EMPLEADO_AN;

	END LOOP;
	CLOSE EMP_ANALISTA;

END;
/
